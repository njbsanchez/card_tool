2. "cd desktop/card_parser_tool"

3. "virtualenv env"

4. "pip3 install -r requirements.txt"

5. "source run_author.txt" or "source run_prod.txt"

1. command + spacebar, search "terminal"

2. "cd desktop/card_parser_tool"

3. "source run_author.txt" or "source run_prod.txt"
